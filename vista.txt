CREATE VIEW comentarios_v
AS SELECT id_chef,id_receta,comentario_opcional,id_plato,comentario_personal,id_menus,comentarioA,comentarioB,comentarioC
FROM comentarios
Where id=1;

AS SELECT id_chef,id_receta,comentario_opcional,id_plato,comentario_personal,id_menus,comentarioA,comentarioB,comentarioC
FROM comentarios
Where id=2;
